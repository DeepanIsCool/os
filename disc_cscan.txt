#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n) {
    int i, j, temp;
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}


int main() {
    int n, head, disk, i;
    int req[100], left[100], right[100];
    int l = 0, r = 0, total = 0;

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request queue: ");
    for (i = 0; i < n; i++) scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &disk);

    right[r++] = disk;
    left[l++] = 0;

    for (i = 0; i < n; i++) {
        if (req[i] >= head) right[r++] = req[i];
        else left[l++] = req[i];
    }

    sort(left, l);
    sort(right, r);

    printf("\nCSCAN Order: %d ", head);

    for (i = 0; i < r; i++) {
        total += abs(right[i] - head);
        head = right[i];
        printf("-> %d ", head);
    }

    head = 0;

    for (i = 0; i < l; i++) {
        total += abs(left[i] - head);
        head = left[i];
        printf("-> %d ", head);
    }

    printf("\nTotal Head Movement = %d\n", total);
    return 0;
}

Disk Scheduling - C-SCAN

Read number of disk requests from the user
Read all request positions
Read initial head position and disk size
Add disk end position to right array and zero to left array
Separate requests into left array for positions less than head and right array for positions greater than or equal to head
Sort both left and right arrays in ascending order
Start from initial head position
Service all requests in right array in ascending order including disk end, calculating total head movement
Jump to beginning of disk at position zero
Service all requests in left array in ascending order, calculating total head movement
