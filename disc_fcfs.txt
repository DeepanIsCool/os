#include <stdio.h>
#include <stdlib.h>

int main() {
    int n, head, i, total = 0;
    int req[100];

    printf("Enter number of requests: ");
    scanf("%d", &n);

    printf("Enter request queue: ");
    for (i = 0; i < n; i++) scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("\nFCFS Order: %d ", head);

    for (i = 0; i < n; i++) {
        total += abs(req[i] - head);
        head = req[i];
        printf("-> %d ", head);
    }

    printf("\nTotal Head Movement = %d\n", total);
    return 0;
}


Disk Scheduling - FCFS

Read number of disk requests from the user
Read all request positions in order
Read initial head position
Start from initial head position
Service each request in the order they arrived
For each request, calculate absolute difference from current head position
Add this difference to total head movement
Move head to the request position
Continue until all requests are serviced
Display the order of disk head movement
Display total head movement
